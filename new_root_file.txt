import sys

def eprint(*args, **kwargs):
    print(*args, file=sys.stderr, **kwargs)
try:
    payload1 = 2.1
    payload2 = 1.2
    budget = 20
    if payload1 == payload2:
        budget -= payload1
    if payload2 != payload1:
        budget -= payload2
    assert budget == 18.8
except Exception as e:
    eprint(e)
